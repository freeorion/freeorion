                    ]
                ]
            ]
            effects = SetTargetResearch value = Value + (NamedReal name = "PLC_NATIVE_APPROPRIATION_RESEARCH_PERPLANET" value = 5.0) *

                    InSystem id = Target.SystemID
                    Planet
                    Unowned
                    Species
                    Homeworld name = LocalCandidate.Species
                ]
    ]
